310 ListNode<T>* head = nullptr;
311public:
312 ~LinkedList() { clear(); }
313 void push_front(const T& val) {
314 ListNode<T>* node = new ListNode<T>(val);
315 node->next = head;
316 head = node;
317 }
318 void push_back(const T& val) {
319 ListNode<T>* node = new ListNode<T>(val);
320 if (!head) { head = node; return; }
321 ListNode<T>* cur = head;
322 while (cur->next) cur = cur->next;
323 cur->next = node;
324 }
325 void pop_front() {
326 if (!head) return;
327 ListNode<T>* tmp = head;
328 head = head->next;
329 delete tmp;
330 }
331 bool empty() const { return head == nullptr; }
332 void clear() { while (!empty()) pop_front(); }
333 void print() const {
334 ListNode<T>* cur = head;
335 while (cur) { println(cur->val); cur = cur->next; }
336 }
337 bool contains(const T& val) const {
338 ListNode<T>* cur = head;
339 while (cur) { if (cur->val == val) return true; cur = cur->next; }
340 return false;
341 }
342 size_t size() const {
343 size_t cnt = 0;
344 ListNode<T>* cur = head;
345 while (cur) { ++cnt; cur = cur->next; }

872// ====================== BASE64 ======================
873string base64_encode(const string& data) {
874 static const string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
875 string result;
876 result.reserve(((data.size() + 2) / 3) * 4);
877 for (size_t i = 0; i < data.size(); i += 3) {
878 unsigned int val = (static_cast<unsigned char>(data[i]) << 16);
879 if (i + 1 < data.size()) val |= (static_cast<unsigned char>(data[i + 1]) << 8);
880 if (i + 2 < data.size()) val |= static_cast<unsigned char>(data[i + 2]);
881 result.push_back(chars[(val >> 18) & 0x3F]);
882 result.push_back(chars[(val >> 12) & 0x3F]);
883 result.push_back(chars[(val >> 6) & 0x3F]);
884 result.push_back(chars[val & 0x3F]);
885 }
886 size_t padding = (3 - (data.size() % 3)) % 3;
887 for (size_t i = 0; i < padding; ++i) result[result.size() - 1 - i] = '=';
888 return result;
889}
890string base64_decode(const string& encoded) {
891 static const string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
892 string result;
893 vector<int> decode(256, -1);
894 for (int i = 0; i < 64; ++i) decode[chars[i]] = i;
895 int val = 0, bits = 0;
896 for (char c : encoded) {
897 if (decode[c] == -1) continue;
898 val = (val << 6) | decode[c];
899 bits += 6;
900 if (bits >= 8) {
901 result.push_back(static_cast<char>((val >> (bits - 8)) & 0xFF));
902 bits -= 8;
903 }
904 }
905 return result;
906}

942template<typename T> void merge(vector<T>& v, int l, int m, int r) {
943 int n1 = m - l + 1, n2 = r - m;
944 vector<T> L(n1), R(n2);
945 for (int i = 0; i < n1; ++i) L[i] = v[l + i];
946 for (int j = 0; j < n2; ++j) R[j] = v[m + 1 + j];
947 int i = 0, j = 0, k = l;
948 while (i < n1 && j < n2) v[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
949 while (i < n1) v[k++] = L[i++];
950 while (j < n2) v[k++] = R[j++];
951}
